name RoleName
roleName) = do
  metadata <- m Metadata
forall (m :: * -> *). MetadataM m => m Metadata
getMetadata
  remoteSchemaMap <- scRemoteSchemas <$> askSchemaCache
  void
    $ onNothing (HashMap.lookup name remoteSchemaMap)
    $ throw400 NotExists
    $ "remote schema "
    <> name

  unless (doesRemoteSchemaPermissionExist metadata name roleName)
    $ throw400 NotExists
